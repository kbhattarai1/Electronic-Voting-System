        public bool Insert(Class2 cl2)
        {
            bool isSucceeded = false;
            string ConnectString = null;
            SqlConnection con = new SqlConnection(ConnectString);
            string sql = "Insert into Vote_cl2(VoterId,Name)";

            try
            {
                con.Open();
                SqlCommand cmd = new SqlCommand(sql, con);
                cmd.Parameters.AddWithValue("@VoterId", cl2.VoterId);
                cmd.Parameters.AddWithValue("@Name", cl2.Name);
                int x = cmd.ExecuteNonQuery();
                if (x > 0)
                {
                    isSucceeded = true;
                }
                else
                {
                    isSucceeded = false;
                }
            }
            catch (Exception ex)
            {

                throw ex;

            }
            finally
            {
                con.Close();
            }
            return isSucceeded;
        }
